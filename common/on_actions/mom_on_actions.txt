# Triggers when the game starts
on_game_start = {
	events = {
		mom_house_divided.1
		mom_severance.1
		mom_breakaway_civ.1
		action.90
	}
}

# A colony has been destroyed. Called just before owner and controller is cleared
# Scope = Planet
on_colony_destroyed = {
	events = {
		mom_breakaway_civ.2
	}
}

#From = Country scope
#This = Planet scope
on_planet_ownerless = {
	events = {
		mom_breakaway_civ.2
	}
}

#Fired whenever a new owner is set for a planet,
#be it after a war or through a trade
#From = Country scope (new owner)
#This = Planet scope
on_planet_transfer = {
	events = {
		mom_breakaway_civ.2
	}
}
